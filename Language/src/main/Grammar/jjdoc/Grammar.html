TOKENS

<DEFAULT> SKIP : {
" "
| "\t"
| "\r"
| "\n"
| <"//" (~["\n","\r"])* ("\n" | "\r" | "\r\n")>
| <"/*" (~["*"])* "*" ("*" | ~["*","/"] (~["*"])* "*")* "/">
}

   
<DEFAULT> TOKEN : {
<DOT: ".">
| <COMMA: ",">
| <ADD: "+">
| <SUB: "-">
| <MULT: "*">
| <DIV: "/">
| <EXCL: "!">
| <TILDE: "~">
| <CONJ: "&&">
| <EAGER_CONJ: "&">
| <DISJ: "||">
| <EAGER_DISJ: "|">
| <EQUALS: "=">
| <ASSIGNMENT: ":=">
| <EQEQ: "==">
| <EXCL_EQUALS: "!=">
| <GREATER: ">">
| <LESS: "<">
| <GR_EQUALS: ">=">
| <LE_EQUALS: "<=">
| <ARROW: "->">
| <SEMICOLON: ";">
| <DOUBLE_SEMICOLON: ";;">
| <COLON: ":">
| <LPAR: "(">
| <RPAR: ")">
| <LCUR: "{">
| <RCUR: "}">
}

   
//Reserved words
    
<DEFAULT> TOKEN : {
<LET: "let">
| <END: "end">
| <IN: "in">
| <TRUE: "true">
| <FALSE: "false">
| <NEW: "new">
| <AS: "as">
| <IF: "if">
| <THEN: "then">
| <ELSE: "else">
| <WHILE: "while">
| <DO: "do">
| <FUN: "fun">
| <INSTANCEOF: "instanceof">
| <PRINTLN: "println">
| <PRINT: "print">
}

   
//Types
    
<DEFAULT> TOKEN : {
<INT: "int">
| <FLOAT: "float">
| <BOOLEAN: "boolean">
| <ANY: "any">
| <VOID: "void">
| <REFERENCE: "ref">
}

   
<DEFAULT> TOKEN : {
<INTEGER_LITERAL: (<DIGIT>)+>
| <FLOAT_LITERAL: (<DIGIT>)+ "." (<DIGIT>)+>
| <IDENTIFIER: <LETTER> (<LETTER> | <DIGIT> | "_")*>
| <#LETTER: ["a"-"z","A"-"Z"]>
| <#DIGIT: ["0"-"9"]>
}

NON-TERMINALS

Start ::= Sequence <DOUBLE_SEMICOLON>
Sequence ::= Statment ( <SEMICOLON> Statment )*
Statment ::= ( Expression | While )
While ::= <WHILE> Sequence <DO> Sequence <END>
Expression ::= Disjunction
Disjunction ::= Conjunction ( ( <DISJ> | <EAGER_DISJ> ) Conjunction )*
Conjunction ::= Equality ( ( <CONJ> | <EAGER_CONJ> ) Equality )*
Equality ::= Comparison ( ( <EQEQ> | <EXCL_EQUALS> ) Comparison )*
Comparison ::= AddSubExpression ( ( <GREATER> | <LESS> | <GR_EQUALS> | <LE_EQUALS> ) AddSubExpression )?
AddSubExpression ::= MultDivExpression ( ( <ADD> | <SUB> ) MultDivExpression )*
MultDivExpression ::= Term ( ( <MULT> | <DIV> ) Term )*
Term ::= AsExpression ( Assignment | ApplyFunction )?
NumberType ::= ( <FLOAT> | <INT> )
AsExpression ::= PrefixUnaryExpression ( <AS> NumberType )?
PrefixUnaryOperator ::= ( <SUB> | <TILDE> )
PrefixUnaryExpression ::= ( PrefixUnaryOperator )* Fact
ApplyFunction ::= <LPAR> FunctionArguments <RPAR>
FunctionArguments ::= ( Sequence ( <COMMA> Sequence )* )?
Assignment ::= <ASSIGNMENT> Expression
Fact ::= ( Constant | Let | IfThenElse | parenthesizedSequence | Identifier | NewReference | Dereference | Function | Println )
FunctionParameters ::= <IDENTIFIER> ( <COMMA> <IDENTIFIER> )*
Function ::= <FUN> FunctionParameters <ARROW> Sequence <END>
IfThenElse ::= <IF> Sequence <THEN> Sequence <ELSE> Sequence <END>
Constant ::= <INTEGER_LITERAL>
| <FLOAT_LITERAL>
| ( <TRUE> | <FALSE> )
NewReference ::= <NEW> Expression
Dereference ::= <EXCL> Fact
PrimitiveType ::= ( NumberType | <BOOLEAN> )
Type ::= ( PrimitiveType | <ANY> | <VOID> )
Identifier ::= <IDENTIFIER>
Binding ::= <IDENTIFIER> <EQUALS> Sequence
BindingsList ::= Binding ( ( <COMMA> )? Binding )*
Let ::= <LET> BindingsList <IN> Sequence <END>
Println ::= ( <PRINTLN> | <PRINT> ) <LPAR> Sequence <RPAR>
parenthesizedSequence ::= <LPAR> Sequence <RPAR>
